interface Plugin {
  meta: {
    name: string
    version: string
    description: string
    author: string
    icon?: string
  }

  config: {
    enabled: boolean
    settings: Record<string, any>
  }

  hooks: {
    onInstall?: () => void
    onUninstall?: () => void
    onActivate?: () => void
    onDeactivate?: () => void
  }

  components?: Record<string, Component>
  stores?: Record<string, Store>
  routes?: RouteRecordRaw[]
}

class PluginManager {
  private plugins: Map<string, Plugin> = new Map()
  private hooks: Map<string, Function[]> = new Map()

  register(plugin: Plugin): void {
    this.plugins.set(plugin.meta.name, plugin)

    // æ³¨å†Œé’©å­
    Object.entries(plugin.hooks).forEach(([hook, handler]) => {
      this.addHook(hook, handler)
    })

    // æ³¨å†Œç»„ä»¶
    if (plugin.components) {
      Object.entries(plugin.components).forEach(([name, component]) => {
        app.component(name, component)
      })
    }

    // æ³¨å†Œè·¯ç”±
    if (plugin.routes) {
      router.addRoute(plugin.routes)
    }
  }

  async activate(pluginName: string): Promise<void> {
    const plugin = this.plugins.get(pluginName)
    if (plugin && plugin.hooks.onActivate) {
      await plugin.hooks.onActivate()
    }
    plugin.config.enabled = true
  }

  async deactivate(pluginName: string): Promise<void> {
    const plugin = this.plugins.get(pluginName)
    if (plugin && plugin.hooks.onDeactivate) {
      await plugin.hooks.onDeactivate()
    }
    plugin.config.enabled = false
  }
}

class DatabaseManager {
  private db: IDBDatabase | null = null

  async init(): Promise<void> {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('NavigatorDB', 1)

      request.onerror = () => reject(request.error)
      request.onsuccess = () => {
        this.db = request.result
        resolve()
      }

      request.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result

        // åˆ›å»ºä¹¦ç­¾å­˜å‚¨
        if (!db.objectStoreNames.contains('bookmarks')) {
          const bookmarkStore = db.createObjectStore('bookmarks', { keyPath: 'id' })
          bookmarkStore.createIndex('categoryId', 'categoryId', { unique: false })
          bookmarkStore.createIndex('tags', 'tags', { unique: false, multiEntry: true })
        }

        // åˆ›å»ºåˆ†ç±»å­˜å‚¨
        if (!db.objectStoreNames.contains('categories')) {
          db.createObjectStore('categories', { keyPath: 'id' })
        }

        // åˆ›å»ºæœç´¢ç´¢å¼•å­˜å‚¨
        if (!db.objectStoreNames.contains('searchIndex')) {
          db.createObjectStore('searchIndex', { keyPath: 'id' })
        }
      }
    })
  }

  async saveBookmark(bookmark: Bookmark): Promise<void> {
    const transaction = this.db!.transaction(['bookmarks'], 'readwrite')
    const store = transaction.objectStore('bookmarks')
    await store.put(bookmark)
  }

  async getBookmarks(): Promise<Bookmark[]> {
    const transaction = this.db!.transaction(['bookmarks'], 'readonly')
    const store = transaction.objectStore('bookmarks')
    return new Promise((resolve) => {
      const request = store.getAll()
      request.onsuccess = () => resolve(request.result)
    })
  }
}

import Fuse from 'fuse.js'

const searchConfig = {
  keys: [
    { name: 'title', weight: 0.7 },
    { name: 'description', weight: 0.2 },
    { name: 'tags', weight: 0.1 },
    { name: 'url', weight: 0.05 }
  ],
  threshold: 0.3,
  includeScore: true,
  includeMatches: true,
  minMatchCharLength: 2,
  useExtendedSearch: true,
  ignoreLocation: true,
  shouldSort: true,
  findAllMatches: false
}

export class SearchManager {
  private fuse: Fuse<Bookmark> | null = null
  private bookmarks: Bookmark[] = []

  async initialize(bookmarks: Bookmark[]): Promise<void> {
    this.bookmarks = bookmarks
    this.fuse = new Fuse(bookmarks, searchConfig)
  }

  async search(query: string, options?: SearchOptions): Promise<SearchResult[]> {
    if (!this.fuse) return []

    let results = this.fuse.search(query)

    // åº”ç”¨è¿‡æ»¤å™¨
    if (options?.categoryId) {
      results = results.filter(result =>
        result.item.categoryId === options.categoryId
      )
    }

    if (options?.tags?.length) {
      results = results.filter(result =>
        options.tags!.some(tag => result.item.tags.includes(tag))
      )
    }

    return results.map(result => ({
      bookmark: result.item,
      score: result.score || 0,
      matches: result.matches || []
    }))
  }

  async updateBookmark(bookmark: Bookmark): Promise<void> {
    // æ›´æ–°ä¹¦ç­¾æ•°æ®
    const index = this.bookmarks.findIndex(b => b.id === bookmark.id)
    if (index >= 0) {
      this.bookmarks[index] = bookmark
      await this.initialize(this.bookmarks)
    }
  }
}
